  /* ── PHOTO UPLOAD ── */
  function handlePhoto(e) {
    const file = e.target.files[0];
    if (!file) return;
    const reader = new FileReader();
    reader.onload = function(ev) {
      const circle = document.getElementById('avatarCircle');
      let img = circle.querySelector('img');
      if (!img) { img = document.createElement('img'); circle.appendChild(img); }
      img.src = ev.target.result;
      document.getElementById('avatarInitials').style.display = 'none';
      showToast('Photo updated');
    };
    reader.readAsDataURL(file);
  }

  function downloadHTML() {
    const clone = document.documentElement.cloneNode(true);
    clone.querySelector('#edit-banner').remove();
    clone.querySelectorAll('.add-btn').forEach(b => b.remove());
    clone.querySelectorAll('.delete-btn').forEach(b => b.remove());
    clone.querySelectorAll('[contenteditable]').forEach(el => el.removeAttribute('contenteditable'));
    clone.querySelector('script') && clone.querySelector('script').remove();
    const html = '<!DOCTYPE html>\n' + clone.outerHTML;
    const blob = new Blob([html], { type: 'text/html' });
    const a = document.createElement('a');
    a.href = URL.createObjectURL(blob);
    a.download = 'resume.html';
    a.click();
    showToast('Downloaded as resume.html');
  }

  }

  function updateInitials() {
    const first = document.querySelectorAll('[contenteditable]')[0].textContent.trim();
    const last = document.querySelectorAll('[contenteditable]')[1].textContent.trim();
    const initials = (first[0] || '') + (last[0] || '');

    if (el && !document.querySelector('#avatarCircle img')) el.textContent = initials.toUpperCase();
  }
